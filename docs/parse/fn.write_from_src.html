Expand description

Parses the grammar specification in src and writes a parser module to fout.

This is equivalent to Rules::new(src) followed by rules.write(fout).

Example

let src = r"
	Start -> Expr;
	Expr -> Fact;
	Fact -> lparen Expr rparen;
	Fact -> n;
	Fact -> plus Fact;
	Fact -> Fact plus n;
";

let mut fout = Vec::<u8>::new();

parse::write_from_src::<u32, u32, _, _>(src, &mut fout).unwrap();
